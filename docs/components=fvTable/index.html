fvTable

This is table component with many features.


Code:

  

    HTML
    
  

  <fvTable :fields="fields" :rows="rows" />
  

  

    Javascript
    
  

  export default {
    data () {
        return {
            fields: ['name', 'family'],
            rows: [
                {
                    name: 'Reza',
                    family: 'Nouri',
                    customField: 'x'
                },
                {
                    name: 'Amir',
                    family: 'Momenian',
                    customField: 'z'
                },
                {
                    name: 'Saeed',
                    family: 'Kamyabi',
                    customField: 'y'
                },
            ]
        }
    }
}
